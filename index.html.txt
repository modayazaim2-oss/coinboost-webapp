    <script>
        // Initialize Telegram WebApp
        let tg = window.Telegram.WebApp;
        
        // Expand to full screen
        tg.ready();
        tg.expand();
        
        // Set header color to match app
        tg.setHeaderColor('#1a1a2e');
        tg.setBackgroundColor('#1a1a2e');
        
        // Get user data
        let user = tg.initDataUnsafe.user;
        
        // Update user info
        if (user) {
            document.getElementById('userName').textContent = user.first_name + (user.last_name ? ' ' + user.last_name : '');
            document.getElementById('userId').textContent = 'ID: ' + user.id;
        }
        
        // Hide loading after 1 second
        setTimeout(() => {
            document.getElementById('loading').classList.add('hidden');
        }, 1000);
        
        // Send data back to bot
        function sendData(action) {
            tg.HapticFeedback.impactOccurred('light');
            
            // Send data to bot
            tg.sendData(JSON.stringify({
                action: action,
                user_id: user ? user.id : null,
                timestamp: new Date().toISOString()
            }));
            
            // Show button press effect
            tg.MainButton.setText('جاري المعالجة...');
            tg.MainButton.show();
        }
        
        // Handle theme changes
        tg.onEvent('themeChanged', function() {
            document.body.style.background = tg.themeParams.bg_color || '#1a1a2e';
        });
    </script>
